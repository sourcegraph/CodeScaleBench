// PolicyAuditLogger provides structured logging of policy evaluation decisions for audit and debugging purposes.
type PolicyAuditLogger struct {
	lock.Mutex
	logger   *slog.Logger
	logLevel slog.Level
}

// NewPolicyAuditLogger creates a new PolicyAuditLogger with the specified logger and log level.
// If logger is nil, a default logger will be used.
func NewPolicyAuditLogger(logger *slog.Logger, level slog.Level) *PolicyAuditLogger {
	if logger == nil {
		logger = slog.Default()
	}
	return &PolicyAuditLogger{
		logger:   logger,
		logLevel: level,
	}
}

// LogDecision logs a policy evaluation decision with identity, policy, verdict, and reason.
// This method is thread-safe and structured for audit purposes.
func (pal *PolicyAuditLogger) LogDecision(identity identity.NumericIdentity, policy string, verdict string, reason string) {
	pal.Lock()
	defer pal.Unlock()

	if pal.logger == nil {
		return
	}

	pal.logger.LogAttrs(context.Background(), pal.logLevel,
		"policy decision",
		slog.String(logfields.Identity, identity.String()),
		slog.String("policy", policy),
		slog.String("verdict", verdict),
		slog.String("reason", reason),
		slog.Int64("timestamp", time.Now().UnixNano()),
	)
}
