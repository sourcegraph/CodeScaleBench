// setupEvaluationServer creates a gRPC gateway mux with the evaluation service
func setupEvaluationServer(t *testing.T, store evaluation.Storer) *runtime.ServeMux {
	ctx := context.Background()
	logger := zaptest.NewLogger(t)
	server := evaluation.New(logger, store)
	mux := runtime.NewServeMux()

	err := rpcevaluation.RegisterEvaluationServiceHandlerServer(ctx, mux, server)
	require.NoError(t, err)

	return mux
}
