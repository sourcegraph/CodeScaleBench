package endpointmanager

import (
	"sync"

	"github.com/cilium/cilium/pkg/endpoint/regeneration"

// EndpointRegenerator is responsible for regenerating endpoints.
type EndpointRegenerator struct {
	// getEndpoints is a callback function to retrieve all endpoints.
	getEndpoints func() []*endpoint.Endpoint
}

// newEndpointRegenerator creates a new EndpointRegenerator instance.
func newEndpointRegenerator(getEndpoints func() []*endpoint.Endpoint) *EndpointRegenerator {
	return &EndpointRegenerator{
		getEndpoints: getEndpoints,
	}
}

// RegenerateAllEndpoints calls a setState for each endpoint and
// regenerates if state transaction is valid. During this process, the endpoint
// list is locked and cannot be modified.
// Returns a waiting group that can be used to know when all the endpoints are
// regenerated.
func (regen *EndpointRegenerator) RegenerateAllEndpoints(regenMetadata *regeneration.ExternalRegenerationMetadata) *sync.WaitGroup {
	var wg sync.WaitGroup

	eps := regen.getEndpoints()
	wg.Add(len(eps))

	// Dereference "reason" field outside of logging statement; see
	// https://github.com/sirupsen/logrus/issues/1003.
	reason := regenMetadata.Reason
	log.WithFields(logrus.Fields{"reason": reason}).Info("regenerating all endpoints")
	for _, ep := range eps {
		go func(ep *endpoint.Endpoint) {
			<-ep.RegenerateIfAlive(regenMetadata)
			wg.Done()
		}(ep)
	}

	return &wg
}

# EndpointRegenerator Extraction - Refactoring Summary

## Objective
Extract regeneration logic from `endpointManager` into a new `EndpointRegenerator` struct in `pkg/endpointmanager/regenerator.go`, following the single-responsibility principle.

## Changes Made

### 1. Created `pkg/endpointmanager/regenerator.go`

- **Constructor**: `newEndpointRegenerator(getEndpoints func() []*endpoint.Endpoint) *EndpointRegenerator`

  - Extracted from `endpointManager`
  - Maintains original functionality:

    - Spawns goroutines to regenerate each endpoint
    - Returns WaitGroup for synchronization
    - Uses package-level `log` for logging

### 2. Updated `pkg/endpointmanager/manager.go`

regenerator *EndpointRegenerator

func (mgr *endpointManager) RegenerateAllEndpoints(regenMetadata *regeneration.ExternalRegenerationMetadata) *sync.WaitGroup {

## Design Decisions

1. **Callback pattern**: Used a callback function for `getEndpoints` instead of passing the manager itself
   - Reduces coupling
   - Makes testing easier
   - Cleaner separation of concerns

   - Encourages internal creation through manager's New() function
   - `EndpointRegenerator` exported (uppercase) for potential future API stability

- No new test files required (functionality unchanged, just extracted)
- Interface contract maintained (EndpointManager still implements RegenerateAllEndpoints)
