type Tracker struct {
	totalEvaluations int64
	matchCount       int64
	errorCount       int64
}

// NewTracker creates a new evaluation metrics tracker.
func NewTracker() *Tracker {
	return &Tracker{}
}

// RecordEvaluation increments the total evaluation counter.
func (t *Tracker) RecordEvaluation() {
	atomic.AddInt64(&t.totalEvaluations, 1)
}

// RecordMatch increments the match counter.
func (t *Tracker) RecordMatch() {
	atomic.AddInt64(&t.matchCount, 1)
}

// RecordError increments the error counter.
func (t *Tracker) RecordError() {
	atomic.AddInt64(&t.errorCount, 1)
}

// TotalEvaluations returns the total number of evaluations processed.
func (t *Tracker) TotalEvaluations() int64 {
	return atomic.LoadInt64(&t.totalEvaluations)
}

// MatchCount returns the number of successful matches.

	return atomic.LoadInt64(&t.matchCount)
}
