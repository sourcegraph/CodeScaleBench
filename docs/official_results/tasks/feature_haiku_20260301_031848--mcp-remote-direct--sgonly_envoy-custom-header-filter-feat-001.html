  // Convert the protobuf config to filter config
  std::vector<HeaderInjectionRule> rules;
  for (const auto& rule : proto_config.rules) {
    rules.emplace_back(HeaderInjectionRule{
        rule.header_name,
        rule.header_value,
        rule.inject_request,
        rule.inject_response,
    });
  }

  ConfigSharedPtr filter_config(std::make_shared<Config>(rules));

  return [filter_config](Http::FilterChainFactoryCallbacks& callbacks) -> void {
    callbacks.addStreamFilter(
        Http::StreamFilterSharedPtr{new CustomHeaderInjectionFilter(filter_config)});
  };
}
