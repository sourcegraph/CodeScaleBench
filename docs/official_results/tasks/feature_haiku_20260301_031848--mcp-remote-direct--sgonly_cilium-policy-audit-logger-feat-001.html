	lock.Mutex
	logLevel logrus.Level
	log      *logrus.Entry
}

func NewPolicyAuditLogger(level logrus.Level) *PolicyAuditLogger {
	return &PolicyAuditLogger{
		logLevel: level,
		log:      auditLog.Logger.WithLevel(level),
	}
}

// LogDecision logs a policy decision with structured fields.
//
// Parameters:
// - identity: The numeric identity of the subject
// - policy: A string identifier for the policy being evaluated
// - verdict: The decision result (e.g., "allow", "deny")
// - reason: A string explaining the decision
func (pal *PolicyAuditLogger) LogDecision(identity identity.NumericIdentity, policy string, verdict string, reason string) {
	pal.Lock()
	defer pal.Unlock()

		logfields.Identity: identity,
		"policy":           policy,
		"verdict":          verdict,
		"reason":           reason,

}
