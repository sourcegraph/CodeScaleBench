func (c *PolicyQuotaController) CheckQuota(namespace string) error {
    c.mu.RLock()
    defer c.mu.RUnlock()
    
    if quota, exists := c.quotaByNamespace[namespace]; exists {
        currentCount := c.policyCountByNamespace[namespace]
        if currentCount >= quota {
            return fmt.Errorf(
                "policy quota exceeded for namespace %q: current=%d, max=%d",

func (c *PolicyQuotaController) handleCNPEvent(
    event resource.Event[*cilium_v2.CiliumNetworkPolicy],
) {
    defer event.Done(nil)  // Proper lifecycle management
    
    cnp := event.Object
    if cnp == nil {
        return
    }
    
    namespace := cnp.GetNamespace()
    c.mu.Lock()
    defer c.mu.Unlock()
    
    switch event.Kind {
    case resource.Upsert:
        c.policyCountByNamespace[namespace]++
        c.logger.WithFields(logrus.Fields{
            "namespace":      namespace,
            "policy":         cnp.GetName(),
            "policyCount":    c.policyCountByNamespace[namespace],
            logfields.LogSubsys: "policy-quota-controller",
        }).Debug("CiliumNetworkPolicy created/updated")
        
    case resource.Delete:
        if count, exists := c.policyCountByNamespace[namespace]; exists && count > 0 {
            c.policyCountByNamespace[namespace]--
        }
    }
}
