## Problem

The current OpenTelemetry instrumentation in Flipt generates all traces with a fixed configuration: it always samples at 100% and applies a predefined set of context propagators. - Propagator setup — where context propagation formats are registered

5. Produce clear error messages for invalid configuration values

// GetPropagators returns a text map propagator based on the configured propagators.
// If no propagators are configured, returns a default composite propagator with TraceContext and Baggage.
func GetPropagators(cfg *config.TracingConfig) propagation.TextMapPropagator {
	if cfg == nil || len(cfg.Propagators) == 0 {
		return propagation.NewCompositeTextMapPropagator(
			propagation.TraceContext{},
			propagation.Baggage{},
		)
	}

	propagators := make([]propagation.TextMapPropagator, 0, len(cfg.Propagators))

	for _, prop := range cfg.Propagators {
		switch prop {
		case "tracecontext":
			propagators = append(propagators, propagation.TraceContext{})
		case "baggage":
			propagators = append(propagators, propagation.Baggage{})
		case "jaeger":
			// Import jaeger propagator if needed
			// For now, we skip jaeger as it requires: go.opentelemetry.io/contrib/propagators/jaeger
		case "xray":
			// Import xray propagator if needed
			// For now, we skip xray as it requires: go.opentelemetry.io/contrib/propagators/aws/xray
		case "ottrace":
			// Import ottrace propagator if needed
			// For now, we skip ottrace as it requires: go.opentelemetry.io/contrib/propagators/ot
		}
	}

	if len(propagators) == 0 {
		return propagation.NewCompositeTextMapPropagator(
			propagation.TraceContext{},
			propagation.Baggage{},
		)
	}

	return propagation.NewCompositeTextMapPropagator(propagators...)
}
