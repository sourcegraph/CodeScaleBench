import time

from django.conf import settings
from django.http import HttpResponseForbidden
from django.utils.deprecation import MiddlewareMixin


class RateLimitMiddleware(MiddlewareMixin):
    """
    Middleware to rate-limit incoming HTTP requests by IP address.

    Tracks requests per client IP and returns a 403 Forbidden response
    when the rate limit is exceeded.
    """

    def __init__(self, get_response):
        super().__init__(get_response)
        self.rate_limit_requests = getattr(settings, "RATE_LIMIT_REQUESTS", 100)
        self.rate_limit_window = getattr(settings, "RATE_LIMIT_WINDOW", 3600)
        self.request_counts = {}  # {ip: [(timestamp, count), ...]}
        self.request_counter = 0

    def process_request(self, request):
        # Clean up expired entries periodically
        self.request_counter += 1
        if self.request_counter >= 100:
            self._cleanup_expired_entries()
            self.request_counter = 0

        # Get client IP address
        ip = self._get_client_ip(request)
        current_time = time.time()

        # Initialize IP entry if not present
        if ip not in self.request_counts:
            self.request_counts[ip] = []

        # Remove timestamps outside the window
        self.request_counts[ip] = [
            ts for ts in self.request_counts[ip]

                f"Rate limit exceeded for IP address {ip}"
