// Format returns a string representation of the plan changes in a compact format.

func (f *CompactDiffFormatter) Format(plan *plans.Plan, schemas *schemarepo.Schemas) string {
	if plan == nil || plan.Changes == nil {
		return ""
	}

	var buf bytes.Buffer

	changes := make(map[plans.Action][]*plans.ResourceInstanceChangeSrc)
	for _, resource := range plan.Changes.Resources {
		if resource.DeposedKey == "" { // Only include current objects, not deposed ones
			changes[resource.Action] = append(changes[resource.Action], resource)
		}
