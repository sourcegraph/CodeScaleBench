                                   struct Curl_easy *data,

  const char *priority_str = data->set.str[STRING_H3_STREAM_PRIORITY];
  if(!priority_str)
    return;

  /* Parse urgency value */
  unsigned char urgency = 3; /* default */
  int incremental = 0;

  /* Try to parse "u=N" format */
  if(priority_str[0] == 'u' && priority_str[1] == '=') {
    char *endptr;
    long val = strtol(priority_str + 2, &endptr, 10);
    if(val >= 0 && val <= 7) {
      urgency = (unsigned char)val;
    }
    /* Check for optional ",i" flag */
    if(endptr && *endptr == ',' && *(endptr + 1) == 'i') {
      incremental = 1;
    }
  }

  /* Store the parsed priority in the data structure for later use */
  data->set.priority.urgency = urgency;
  data->set.priority.incremental = incremental ? TRUE : FALSE;
}
