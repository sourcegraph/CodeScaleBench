// PolicyAuditLogger provides structured logging of policy evaluation decisions
// for audit and debugging purposes.
type PolicyAuditLogger struct {
	lock.Mutex
	logLevel logrus.Level
}

// NewPolicyAuditLogger creates a new PolicyAuditLogger with the specified log level.
func NewPolicyAuditLogger(logLevel logrus.Level) *PolicyAuditLogger {
	return &PolicyAuditLogger{
		logLevel: logLevel,
	}
}

// LogDecision logs a policy decision with identity, policy, verdict, and reason.
//
// This method is thread-safe and logs structured information about policy
// decisions made during policy evaluation.
func (pal *PolicyAuditLogger) LogDecision(identity identity.NumericIdentity, policy string, verdict api.Decision, reason string) {
	pal.Lock()
	defer pal.Unlock()

	auditLog.WithFields(logrus.Fields{
		logfields.Identity: identity,
		"policy":           policy,
		"verdict":          verdict.String(),
		"reason":           reason,
	}).Log(pal.logLevel, "Policy decision logged")
}
