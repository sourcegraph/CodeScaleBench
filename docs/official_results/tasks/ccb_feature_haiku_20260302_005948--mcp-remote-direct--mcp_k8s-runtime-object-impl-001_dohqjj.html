Find all concrete struct types that implement the `runtime.Object` interface across Kubernetes API repositories.

## Background

The `runtime.Object` interface is the foundational interface for all Kubernetes API types. It is defined in the `k8s.io/apimachinery/pkg/runtime` package and requires two methods:

```go
type Object interface {
    GetObjectKind() schema.ObjectKind
    DeepCopyObject() Object
}
```

Every Kubernetes resource type (Pod, Deployment, Service, etc.) implements this interface by embedding `metav1.TypeMeta` and implementing the autogenerated `DeepCopyObject()` method.

## Repositories

Two repositories are available under `/workspace/`:

- `/workspace/api/` — kubernetes/api (standard API types: core/v1, apps/v1, etc.)
- `/workspace/apimachinery/` — kubernetes/apimachinery (runtime machinery, metav1, schema)

## Task

Find **all struct types** that implement the `runtime.Object` interface across both repositories. Focus on:

1. **Concrete resource types** (Pod, Deployment, Service, ConfigMap, etc.)
2. **Types with TypeMeta embedded** — the standard pattern for runtime.Object implementors
3. **Types with DeepCopyObject() methods** — either explicit or autogenerated

package main

import (
	"fmt"
	"go/ast"
	"go/parser"
	"go/token"
	"os"
	"path/filepath"
	"sort"
	"strings"
)

type TypeInfo struct {
	Repo   string
	File   string
	Name   string
	HasTypeMetaField bool
}

func main() {
	var results []TypeInfo
	
	// Process both repositories
	repos := map[string]string{
		"kubernetes/api":         "/tmp/api_repo",
		"kubernetes/apimachinery": "/tmp/apimachinery_repo",
	}

	for repoName, repoPath := range repos {
		walkDir(repoPath, repoName, &results)
	}

	// Sort and deduplicate
	sort.Slice(results, func(i, j int) bool {
		if results[i].File != results[j].File {
			return results[i].File < results[j].File
		}
		return results[i].Name < results[j].Name
	})

	// Print as JSON-like format
	for _, r := range results {
		if r.HasTypeMetaField && !strings.Contains(r.Name, "List") && r.Repo == "kubernetes/api" {
			fmt.Printf("{\"repo\": \"%s\", \"file\": \"%s\", \"struct_name\": \"%s\"}\n", r.Repo, r.File, r.Name)
		}
	}
}

func walkDir(root, repoName string, results *[]TypeInfo) {
	filepath.Walk(root, func(path string, info os.FileInfo, err error) error {
		if err != nil || info.IsDir() {
			return nil
		}
		
		if !strings.HasSuffix(path, "types.go") || strings.Contains(path, "zz_generated") {
			return nil
		}

		extractStructs(path, repoName, root, results)
		return nil
	})
}

func extractStructs(filePath, repoName, root string, results *[]TypeInfo) {
	fset := token.NewFileSet()
	file, err := parser.ParseFile(fset, filePath, nil, 0)
	if err != nil {
		return
	}

	relPath := strings.TrimPrefix(filePath, root+"/")

	for _, decl := range file.Decls {
		genDecl, ok := decl.(*ast.GenDecl)
		if !ok || genDecl.Tok != token.TYPE {
			continue
		}

		for _, spec := range genDecl.Specs {
			typeSpec := spec.(*ast.TypeSpec)
			structType, ok := typeSpec.Type.(*ast.StructType)
			if !ok {
				continue
			}

			hasTypeMeta := hasTypeMetaField(structType)
			*results = append(*results, TypeInfo{
				Repo:             repoName,
				File:             relPath,
				Name:             typeSpec.Name.Name,
				HasTypeMetaField: hasTypeMeta,
			})
		}
	}
}

func hasTypeMetaField(structType *ast.StructType) bool {
	if structType.Fields == nil {
		return false
	}

	for _, field := range structType.Fields.List {
		if field.Names == nil || len(field.Names) == 0 {
			// Embedded field
			if ident, ok := field.Type.(*ast.Ident); ok && ident.Name == "TypeMeta" {
				return true
			}
			// Check for metav1.TypeMeta
			if sel, ok := field.Type.(*ast.SelectorExpr); ok {
				if sel.Sel.Name == "TypeMeta" {
					return true
				}
			}
		}
	}
	return false
}
