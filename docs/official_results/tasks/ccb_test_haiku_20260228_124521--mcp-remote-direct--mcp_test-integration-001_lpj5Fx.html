// setupServer creates an evaluation server and returns a test HTTP server.
func setupServer(t *testing.T, store evaluation.Storer) *httptest.Server {
	logger := zaptest.NewLogger(t)
	evalServer := evaluation.New(logger, store)

	// Create a grpc server with the evaluation service
	grpcServer := grpc.NewServer()
	evalServer.RegisterGRPC(grpcServer)

	// Create the HTTP test server with the gRPC server running
	lis, err := net.Listen("tcp", "127.0.0.1:0")
	require.NoError(t, err)

	go grpcServer.Serve(lis)
	t.Cleanup(func() {
		grpcServer.Stop()
		lis.Close()
	})

	// Get the actual address the gRPC server is listening on
	grpcAddr := lis.Addr().String()

	// Create a client connection to the gRPC server
	conn, err := grpc.Dial(
		grpcAddr,
		grpc.WithInsecure(),
		grpc.WithBlock(),
	)
	require.NoError(t, err)
	t.Cleanup(func() {
		conn.Close()
	})

	// Register the gRPC gateway handlers
	mux := grpc.NewMux()
	err = rpcevaluation.RegisterEvaluationServiceHandler(context.Background(), mux, conn)
	require.NoError(t, err)

	// Create HTTP test server
	server := httptest.NewServer(mux)
	t.Cleanup(server.Close)

	return server
}

// setupServer creates an evaluation server and returns a test HTTP server.
func setupServer(t *testing.T, store evaluation.Storer) *httptest.Server {
	logger := zaptest.NewLogger(t)
	evalServer := evaluation.New(logger, store)

	// Create the gRPC gateway mux
	mux := runtime.NewServeMux()

	// Register the evaluation service handlers directly with the server
	err := rpcevaluation.RegisterEvaluationServiceHandlerServer(context.Background(), mux, evalServer)
	require.NoError(t, err)

	// Create HTTP test server
	server := httptest.NewServer(mux)
	t.Cleanup(server.Close)

	return server
}
