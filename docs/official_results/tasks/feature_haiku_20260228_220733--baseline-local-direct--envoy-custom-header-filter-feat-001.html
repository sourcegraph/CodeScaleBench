#include "source/extensions/filters/http/custom_header_injection/filter.h"

namespace Envoy {
namespace Extensions {
namespace HttpFilters {
namespace CustomHeaderInjection {

Http::FilterHeadersStatus
CustomHeaderInjectionFilter::decodeHeaders(Http::RequestHeaderMap& headers, bool) {
  // Inject headers into request
  for (const auto& rule : config_->rules()) {
    if (rule.inject_on_request) {
      headers.addCopy(Http::LowerCaseString(rule.header_name), rule.header_value);
    }
  }
  return Http::FilterHeadersStatus::Continue;
}

Http::FilterHeadersStatus
CustomHeaderInjectionFilter::encodeHeaders(Http::ResponseHeaderMap& headers, bool) {
  // Inject headers into response
  for (const auto& rule : config_->rules()) {
    if (rule.inject_on_response) {
      headers.addCopy(Http::LowerCaseString(rule.header_name), rule.header_value);
    }
  }
  return Http::FilterHeadersStatus::Continue;
}
