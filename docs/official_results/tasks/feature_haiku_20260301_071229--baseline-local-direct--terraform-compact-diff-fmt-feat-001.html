// for terraform plan, showing only resource-level changes without

// Format returns a compact summary of resource changes in a plan.
// The output groups changes by action type and includes a summary count

func (f *CompactDiffFormatter) Format(plan *plans.Plan, schemas *schemarepo.Schemas) string {
	if plan == nil || plan.Changes == nil || plan.Changes.Resources == nil {
		return ""
	}

	changes := make(map[plans.Action][]*plans.ResourceInstanceChangeSrc)

		if rc.DeposedKey != states.NotDeposed {
			// Skip deposed resources for compact output
			continue
