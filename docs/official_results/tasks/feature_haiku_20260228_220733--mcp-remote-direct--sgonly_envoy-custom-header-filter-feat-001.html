/**
 * Parse header injection rules from configuration.
 * Format: "header_name:header_value;header_name:header_value;..."
 * Example: "X-Custom-Header:custom-value;X-Another:another-value"
 */
Config::HeaderInjectionRules parseRules(const std::string& rules_str) {
  Config::HeaderInjectionRules rules;

  if (rules_str.empty()) {
    return rules;
  }

  size_t pos = 0;
  while (pos < rules_str.length()) {
    // Find the next semicolon (or end of string)
    size_t next_semi = rules_str.find(';', pos);
    if (next_semi == std::string::npos) {
      next_semi = rules_str.length();
    }

    // Extract the rule string
    std::string rule_str = rules_str.substr(pos, next_semi - pos);

    // Find the colon separator
    size_t colon_pos = rule_str.find(':');
    if (colon_pos != std::string::npos) {
      std::string header_name = rule_str.substr(0, colon_pos);
      std::string header_value = rule_str.substr(colon_pos + 1);

      if (!header_name.empty() && !header_value.empty()) {
        rules.emplace_back(Config::HeaderInjectionRule{header_name, header_value});
      }
    }

    pos = next_semi + 1;
  }

  return rules;
}
