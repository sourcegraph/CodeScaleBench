from django.conf import settings
from django.core.cache import cache
from django.http import HttpResponseTooManyRequests
from django.utils.deprecation import MiddlewareMixin


class RateLimitMiddleware(MiddlewareMixin):
    """
    Middleware that provides configurable per-IP request rate limiting
    using Django's cache framework.
    """

    def process_request(self, request):
        # Get rate limit settings
        rate_limit_requests = getattr(settings, "RATE_LIMIT_REQUESTS", 100)
        rate_limit_window = getattr(settings, "RATE_LIMIT_WINDOW", 3600)

        # Get the client IP address
        ip_address = self.get_client_ip(request)

        # Create a cache key for this IP
        cache_key = f"rate_limit:{ip_address}"

        # Get current request count
        request_count = cache.get(cache_key, 0)

        # Check if limit exceeded
        if request_count >= rate_limit_requests:
            return HttpResponseTooManyRequests()

        # Increment the counter and set it with the window timeout
        cache.set(cache_key, request_count + 1, rate_limit_window)

        return None

    def get_client_ip(self, request):
        """
        Get the client's IP address from the request.
        Handles X-Forwarded-For header for proxied requests.
        """
        x_forwarded_for = request.META.get("HTTP_X_FORWARDED_FOR")
        if x_forwarded_for:
            # Take the first IP if multiple are present
            ip = x_forwarded_for.split(",")[0].strip()
        else:
            ip = request.META.get("REMOTE_ADDR")
        return ip or "unknown"
