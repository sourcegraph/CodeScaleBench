// PolicyAuditLogger provides structured logging of policy evaluation decisions
// for audit and debugging purposes.
type PolicyAuditLogger struct {
	lock.Mutex

	logger logrus.FieldLogger
	level  logrus.Level
}

// NewPolicyAuditLogger creates a new PolicyAuditLogger with the specified log level.
func NewPolicyAuditLogger(level logrus.Level) *PolicyAuditLogger {
	return &PolicyAuditLogger{
		logger: logging.DefaultLogger,
		level:  level,
	}
}

// NewPolicyAuditLoggerWithLogger creates a new PolicyAuditLogger with the specified
// logger and log level.
func NewPolicyAuditLoggerWithLogger(logger logrus.FieldLogger, level logrus.Level) *PolicyAuditLogger {
	return &PolicyAuditLogger{
		logger: logger,
		level:  level,
	}
}

// LogDecision logs a policy decision with identity, policy information, verdict and reason.
func (pal *PolicyAuditLogger) LogDecision(identity identity.NumericIdentity, policy string, verdict string, reason string) {
	pal.Lock()
	defer pal.Unlock()

	if pal.logger.(*logrus.Entry).Logger.Level < pal.level {
		return
	}

	entry := pal.logger.WithFields(logrus.Fields{
		logfields.Identity: identity,
		"policy":           policy,
		"verdict":          verdict,
		"reason":           reason,
	})

	entry.Log(pal.level, "Policy decision")
}
