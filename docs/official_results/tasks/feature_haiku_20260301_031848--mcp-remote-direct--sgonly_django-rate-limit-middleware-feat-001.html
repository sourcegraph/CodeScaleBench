from django.conf import settings
from django.core.cache import cache
from django.http import HttpResponseTooManyRequests
from django.utils.deprecation import MiddlewareMixin


class RateLimitMiddleware(MiddlewareMixin):
    """
    Middleware to implement per-IP request rate limiting using Django's cache framework.

    Configuration:
    - RATE_LIMIT_REQUESTS: Maximum number of requests per IP (default: 100)
    - RATE_LIMIT_WINDOW: Time window in seconds (default: 3600)
    """

    def process_request(self, request):
        """
        Check if the client IP has exceeded the rate limit.
        Returns a 429 (Too Many Requests) response if limit is exceeded.
        """
        # Get configuration from settings with defaults
        rate_limit_requests = getattr(settings, 'RATE_LIMIT_REQUESTS', 100)
        rate_limit_window = getattr(settings, 'RATE_LIMIT_WINDOW', 3600)

        # Get client IP address
        client_ip = request.META.get('REMOTE_ADDR', '')

        # Create cache key unique to this IP
        cache_key = f'ratelimit:{client_ip}'

        # Get current request count for this IP
        request_count = cache.get(cache_key, 0)

        # Check if limit is exceeded
        if request_count >= rate_limit_requests:
            return HttpResponseTooManyRequests('Rate limit exceeded')

        # Increment request count
        cache.set(cache_key, request_count + 1, rate_limit_window)

        return None
