from django.conf import settings
from django.core.cache import cache
from django.http import HttpResponseTooManyRequests
from django.utils.deprecation import MiddlewareMixin


class RateLimitMiddleware(MiddlewareMixin):
    """
    Middleware that implements per-IP request rate limiting using Django's cache framework.

    Configuration via Django settings:
    - RATE_LIMIT_REQUESTS: Maximum number of requests allowed per IP (default: 100)
    - RATE_LIMIT_WINDOW: Time window in seconds (default: 3600)
    """

    def __init__(self, get_response):
        super().__init__(get_response)
        self.requests_limit = getattr(settings, "RATE_LIMIT_REQUESTS", 100)
        self.window_seconds = getattr(settings, "RATE_LIMIT_WINDOW", 3600)

    def get_client_ip(self, request):
        """
        Get the client IP address from the request.
        """
        return request.META.get("REMOTE_ADDR", "")

    def process_request(self, request):
        """
        Check if the client has exceeded the rate limit.
        If limit is exceeded, return a 429 response.
        Otherwise, increment the request counter.
        """
        client_ip = self.get_client_ip(request)
        if not client_ip:
            # If we can't determine the IP, allow the request
            return None

        cache_key = f"ratelimit_{client_ip}"

        # Get current request count
        try:
            request_count = cache.incr(cache_key)
        except ValueError:
            # Key doesn't exist, create it with value 1
            cache.set(cache_key, 1, self.window_seconds)
            request_count = 1

        # Check if limit is exceeded
        if request_count > self.requests_limit:
            return HttpResponseTooManyRequests()

        return None
