import os
import subprocess
import re
from collections import defaultdict

# Directories to analyze
scalar_dir = "/workspace/llvm-project--a8f3c97d/llvm/lib/Transforms/Scalar"
ipo_dir = "/workspace/llvm-project--a8f3c97d/llvm/lib/Transforms/IPO"

files_by_dir = {
    "scalar": scalar_dir,
    "ipo": ipo_dir
}

# Pattern to search for
patterns = {
    "INITIALIZE_PASS": r"INITIALIZE_PASS\(",
    "INITIALIZE_PASS_BEGIN": r"INITIALIZE_PASS_BEGIN\(",
    "INITIALIZE_PASS_END": r"INITIALIZE_PASS_END\(",
    "FunctionPass": r":\s*public\s+FunctionPass",
    "ModulePass": r":\s*public\s+ModulePass",
    "create_pass": r"^\s*\w+Pass\*\s+create\w+Pass\(",
}

results = defaultdict(lambda: {"patterns_found": set(), "repo": "sg-evals/llvm-project--a8f3c97d"})

def check_file(filepath, relpath):
    try:
        with open(filepath, 'r') as f:
            content = f.read()
    except:
        return
    
    for pattern_name, pattern in patterns.items():
        if re.search(pattern, content, re.MULTILINE):
            results[relpath]["patterns_found"].add(pattern_name)

# Check Scalar files
for filename in os.listdir(scalar_dir):
    if filename.endswith(".cpp"):
        filepath = os.path.join(scalar_dir, filename)
        relpath = f"llvm/lib/Transforms/Scalar/{filename}"
        check_file(filepath, relpath)

# Check IPO files
for filename in os.listdir(ipo_dir):
    if filename.endswith(".cpp"):
        filepath = os.path.join(ipo_dir, filename)
        relpath = f"llvm/lib/Transforms/IPO/{filename}"
        check_file(filepath, relpath)

# Output results
files_with_patterns = []
for relpath, data in sorted(results.items()):
    if data["patterns_found"]:
        files_with_patterns.append({
            "path": relpath,
            "repo": data["repo"],
            "patterns": sorted(list(data["patterns_found"]))
        })

print(f"Found {len(files_with_patterns)} files with legacy patterns:\n")
for item in files_with_patterns:
    print(f"{item['path']}: {', '.join(item['patterns'])}")

# Create the answer JSON
answer = {
    "files": [
        {"repo": item["repo"], "path": item["path"]}
        for item in files_with_patterns
    ]
}

import json
with open("/workspace/answer.json", "w") as f:
    json.dump(answer, f, indent=2)

print(f"\n\nWrote results to /workspace/answer.json")
EOF
