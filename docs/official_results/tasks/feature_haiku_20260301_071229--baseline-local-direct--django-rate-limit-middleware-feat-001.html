configurable per-IP request rate limiting using Django's cache framework.

## Requirements

1. **Create `django/middleware/ratelimit.py`** with a `RateLimitMiddleware` class that:
   - Inherits from `MiddlewareMixin` (see `django/utils/deprecation.py`)

"""
Rate limiting middleware for Django.

Provides per-IP request rate limiting using Django's cache framework.
"""
from django.conf import settings
from django.core.cache import cache
from django.http import HttpResponseTooManyRequests
from django.utils.deprecation import MiddlewareMixin


class RateLimitMiddleware(MiddlewareMixin):
    """
    Middleware that rate limits requests based on client IP address.

    Uses Django's cache framework to track request counts per IP.
    Returns 429 (Too Many Requests) when a client exceeds the configured limit.

    Configuration via settings:
    - RATE_LIMIT_REQUESTS: Maximum requests per window (default: 100)
    - RATE_LIMIT_WINDOW: Time window in seconds (default: 3600)
    """

    def __init__(self, get_response):
        super().__init__(get_response)
        self.rate_limit_requests = getattr(settings, "RATE_LIMIT_REQUESTS", 100)
        self.rate_limit_window = getattr(settings, "RATE_LIMIT_WINDOW", 3600)

    def process_request(self, request):
        """
        Check if the client has exceeded the rate limit.

        Returns a 429 response if the limit is exceeded, None otherwise.
        """
        # Get the client's IP address
        client_ip = request.META.get("REMOTE_ADDR", "")

        # Create a cache key for this IP
        cache_key = f"ratelimit:{client_ip}"

        # Get the current request count
        request_count = cache.get(cache_key, 0)

        # Check if the limit has been exceeded
        if request_count >= self.rate_limit_requests:
            return HttpResponseTooManyRequests("Rate limit exceeded")

        # Increment the request count and set the timeout
        cache.set(cache_key, request_count + 1, self.rate_limit_window)

        # Allow the request to proceed
        return None
