// SilenceInput is the structure for silence input in the bulk endpoint.
type SilenceInput struct {
	Matchers []SilenceMatcher `json:"matchers"`
	StartsAt time.Time        `json:"startsAt"`
	EndsAt   time.Time        `json:"endsAt"`
	Comment  string           `json:"comment"`
}

// SilenceMatcher represents a label matcher for a silence.
type SilenceMatcher struct {
	Name  string `json:"name"`
	Value string `json:"value"`
	IsRe  bool   `json:"isRe,omitempty"`
}

// SilenceOutput is the structure for silence output (created silence).
type SilenceOutput struct {
	ID string `json:"id"`
}

// CreateSilence creates a silence and returns its ID.
func (m *mockSilenceManager) CreateSilence(ctx context.Context, matchers []*labels.Matcher, startsAt, endsAt time.Time, createdBy, comment string) (string, error) {
	m.counter++
	id := "silence-" + string(rune('a'+m.counter-1))

	// Store matchers for verification
	smatchersSlice := make([]silenceMatcher, len(matchers))
	for i, m := range matchers {
		smatchersSlice[i] = silenceMatcher{
			Name:  m.Name,
			Value: m.Value,
			IsRe:  m.Type == labels.MatchRegexp,
		}
	}

	m.silences[id] = silenceInput{
		Matchers: smatchersSlice,
		StartsAt: startsAt,
		EndsAt:   endsAt,
		Comment:  comment,
	}
	return id, nil
}
