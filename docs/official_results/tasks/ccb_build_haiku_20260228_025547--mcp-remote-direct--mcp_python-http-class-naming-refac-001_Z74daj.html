            with open(filepath, 'r') as f:
                content = f.read()
            
            # Replace patterns carefully
            # 1. class Request( -> class HTTPRequest(
            content = re.sub(r'\bclass Request\(', 'class HTTPRequest(', content)
            
            # 2. from .wrappers import Request -> from .wrappers import HTTPRequest
            content = re.sub(r'from \.wrappers import Request\b', 'from .wrappers import HTTPRequest', content)
            
            # 3. from flask import Request -> from flask import HTTPRequest
            content = re.sub(r'from flask import Request\b', 'from flask import HTTPRequest', content)
            
            # 4. Type hints: Request[...] -> HTTPRequest[...]
            content = re.sub(r'\bRequest\[', 'HTTPRequest[', content)
            
            # 5. Type hints: : Request -> : HTTPRequest
            content = re.sub(r': Request\b', ': HTTPRequest', content)
            
            # 6. Type hints: -> Request -> -> HTTPRequest
            content = re.sub(r'-> Request\b', '-> HTTPRequest', content)
            
            # 7. Instantiation: Request( -> HTTPRequest(
            content = re.sub(r'\bRequest\(', 'HTTPRequest(', content)
            
            # 8. type[Request] -> type[HTTPRequest]
            content = re.sub(r'type\[Request\]', 'type[HTTPRequest]', content)
            
            # 9. request_class: type[Request] = Request
            content = re.sub(r'request_class: type\[HTTPRequest\] = HTTPRequest', 'request_class: type[HTTPRequest] = HTTPRequest', content)
            
            # 10. Docstring references in code: :class:`~flask.Request` -> :class:`~flask.HTTPRequest`
            content = re.sub(r':class:`~flask\.Request`', ':class:`~flask.HTTPRequest`', content)
            
            # 11. as Request -> as HTTPRequest
            content = re.sub(r' as Request\b', ' as HTTPRequest', content)
            
            # 12. Union references: Union[Request, - > Union[HTTPRequest, 
            content = re.sub(r'\| Request\b', '| HTTPRequest', content)
            
            with open(filepath, 'w') as f:
                f.write(content)

print("Flask Request renaming completed")
