// CompactDiffFormatter formats a plan as a condensed summary, showing only
// resource-level changes without attribute details.
type CompactDiffFormatter struct{}

// Format takes a plan and schemas and returns a compact diff summary.
// It groups changes by action type and shows a summary count at the end.
func (c *CompactDiffFormatter) Format(plan *plans.Plan, schemas *schemarepo.Schemas) string {
	if plan == nil || plan.Changes == nil {
		return ""
	}

	// Group changes by action
	changes := make(map[plans.Action][]*plans.ResourceInstanceChangeSrc)

	for _, rc := range plan.Changes.Resources {
		changes[rc.Change.Action] = append(changes[rc.Change.Action], rc)
	}

	// Define the order we want to display actions
	actionOrder := []plans.Action{
		plans.Create,
		plans.Update,
		plans.Delete,
		plans.DeleteThenCreate,
		plans.CreateThenDelete,
		plans.Read,
		plans.Forget,
		plans.CreateThenForget,
		plans.NoOp,
	}

	var output strings.Builder
	var totalCounts map[plans.Action]int = make(map[plans.Action]int)

	// Process each action in order
	for _, action := range actionOrder {
		resourceChanges, exists := changes[action]
		if !exists || len(resourceChanges) == 0 {
			continue
		}
