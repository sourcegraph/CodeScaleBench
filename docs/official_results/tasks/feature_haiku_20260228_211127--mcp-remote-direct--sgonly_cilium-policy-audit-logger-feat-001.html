// PolicyAuditLogger provides structured logging of policy evaluation decisions
// for audit and debugging purposes.
type PolicyAuditLogger struct {
	lock.Mutex

	// log is the logger instance used for audit logging
	log logrus.FieldLogger

	// logLevel is the minimum log level for policy audit events
	logLevel logrus.Level

	// output is an optional custom output writer for audit logs
	// If set, audit logs are written as JSON to this writer in addition to the logger
	output io.Writer
}

// NewPolicyAuditLogger creates a new PolicyAuditLogger with the specified configuration.
// If output is nil, logs are written only to the standard logger.
func NewPolicyAuditLogger(logLevel logrus.Level, output io.Writer) *PolicyAuditLogger {
	logger := logging.DefaultLogger.WithField(logfields.LogSubsys, AuditLogSubsys)
	return &PolicyAuditLogger{
		log:      logger,
		logLevel: logLevel,
		output:   output,
	}
}
