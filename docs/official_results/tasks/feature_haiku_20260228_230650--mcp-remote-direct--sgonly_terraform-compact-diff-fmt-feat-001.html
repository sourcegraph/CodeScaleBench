// CompactDiffFormatter produces a condensed, summary-style diff output
// for terraform plan, showing only resource-level changes without
// attribute-level detail.
type CompactDiffFormatter struct{}

// Format returns a compact summary of the planned changes grouped by action type.
// The output shows one line per resource change in the format:
//   [action] resource_type.name
// followed by a summary count at the end.
func (c *CompactDiffFormatter) Format(plan *plans.Plan, schemas *terraform.Schemas) string {
	if plan == nil || plan.Changes == nil {
		return ""
	}

	var buf bytes.Buffer

	// Group changes by action
	actionGroups := make(map[plans.Action][]*plans.ResourceInstanceChangeSrc)
	for _, rc := range plan.Changes.Resources {
		// Only include non-NoOp changes
		if rc.Action != plans.NoOp {
			actionGroups[rc.Action] = append(actionGroups[rc.Action], rc)
		}
	}

	// Define the order to display actions
	actionOrder := []plans.Action{
		plans.Create,
		plans.Update,
		plans.Delete,
		plans.DeleteThenCreate,
		plans.CreateThenDelete,
		plans.Read,
		plans.Forget,
		plans.CreateThenForget,
	}

	// Track total counts
	totalCounts := make(map[plans.Action]int)

	// Print changes grouped by action in the defined order
	for _, action := range actionOrder {
		changes := actionGroups[action]
		if len(changes) == 0 {
			continue
		}
