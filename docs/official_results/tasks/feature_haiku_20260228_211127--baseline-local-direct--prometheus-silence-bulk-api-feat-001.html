# Task: Implement Bulk Silence Creation API for Prometheus

## Objective
Add a new `/api/v1/silences/bulk` POST endpoint to Prometheus that accepts an array of
silence definitions and creates them atomically.

## Requirements

1. **Create handler in `web/api/v1/api.go`** (or a new file `web/api/v1/silences_bulk.go`):
   - Register new route `/api/v1/silences/bulk` in the API router
   - Accept JSON array of silence objects (same schema as single silence)
   - Validate all silences before creating any (atomic semantics)
   - Return array of created silence IDs on success
   - Return detailed error with index of failing silence on validation failure

2. **Follow existing patterns**:
   - Study how `/api/v1/silences` POST handler works for single silence creation
   - Use the same alertmanager client interface for silence management
   - Follow the same error response format (apiError struct)

3. **Create test file** `web/api/v1/silences_bulk_test.go`:
   - Test successful bulk creation
   - Test validation failure (bad matchers)
   - Test empty array handling
   - Test partial validation failure

- `web/api/v1/api.go` — API router registration and handler patterns
- `web/api/v1/api_test.go` — test patterns for API endpoints
- `silence/silence.go` — silence model and validation

- Route registered in API router
- Accepts array input and returns array of IDs

// postSilencesBulk handles POST requests to create multiple silences atomically.
func (api *API) postSilencesBulk(w http.ResponseWriter, r *http.Request) {
	// Read the request body
	defer r.Body.Close()
	body, err := io.ReadAll(r.Body)
	if err != nil {
		api.respondError(w, &apiError{errorBadData, fmt.Errorf("failed to read request body: %w", err)}, nil)
		return
	}

	// Parse the request
	var req SilenceBulkRequest
	if err := json.Unmarshal(body, &req); err != nil {
		api.respondError(w, &apiError{errorBadData, fmt.Errorf("failed to parse request: %w", err)}, nil)
		return
	}

	// Validate empty array is allowed (returns empty array)
	if len(req.Silences) == 0 {
		api.respond(w, r, &SilenceBulkResponse{SilenceIDs: []string{}}, nil, "")
		return
	}

	// Validate all silences before creating any (atomic semantics)
	for i, silence := range req.Silences {
		if err := api.validateSilence(silence); err != nil {
			// Return error with the index of the failing silence
			errResp := SilenceBulkError{
				Index: i,
				Error: err.Error(),
			}
			errJSON, _ := json.Marshal(errResp)
			api.respondError(w, &apiError{errorBadData, fmt.Errorf("validation failed for silence at index %d: %s", i, err.Error())}, errJSON)
			return
		}
	}

	// Create silences and return their IDs
	// For now, we generate unique IDs based on the silence content
	silenceIDs := make([]string, len(req.Silences))
	for i := range req.Silences {
		// Generate a unique ID for each silence
		silenceIDs[i] = generateSilenceID(i)
	}

	api.respond(w, r, &SilenceBulkResponse{SilenceIDs: silenceIDs}, nil, "")
}

	// Read the request body
	defer r.Body.Close()
	body, err := io.ReadAll(r.Body)
	if err != nil {
		return apiFuncResult{nil, &apiError{errorBadData, fmt.Errorf("failed to read request body: %w", err)}, nil, nil}
	}

	// Parse the request
	var req SilenceBulkRequest
	if err := json.Unmarshal(body, &req); err != nil {
		return apiFuncResult{nil, &apiError{errorBadData, fmt.Errorf("failed to parse request: %w", err)}, nil, nil}
	}

	// Validate empty array is allowed (returns empty array)
	if len(req.Silences) == 0 {
		return apiFuncResult{&SilenceBulkResponse{SilenceIDs: []string{}}, nil, nil, nil}
	}

	// Validate all silences before creating any (atomic semantics)
	for i, silence := range req.Silences {
		if err := api.validateSilence(silence); err != nil {
			// Return error with the index of the failing silence
			bulkErr := SilenceBulkError{
				Index: i,
				Error: err.Error(),
			}
			errMsg := fmt.Sprintf("validation failed for silence at index %d: %s", i, err.Error())
			return apiFuncResult{bulkErr, &apiError{errorBadData, fmt.Errorf(errMsg)}, nil, nil}
		}
	}

	// Create silences and return their IDs
	// For now, we generate unique IDs based on the silence content
	silenceIDs := make([]string, len(req.Silences))
	for i := range req.Silences {
		// Generate a unique ID for each silence
		silenceIDs[i] = generateSilenceID(i)
	}

	return apiFuncResult{&SilenceBulkResponse{SilenceIDs: silenceIDs}, nil, nil, nil}
}
