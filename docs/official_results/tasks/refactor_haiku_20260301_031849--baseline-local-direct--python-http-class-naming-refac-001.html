import os
import re

def fix_flask_request_references(filepath, content):
    """Fix remaining Request references in Flask files"""
    
    # Check if this file imports Request
    if "from .wrappers import Request" in content:
        # Replace the import
        content = content.replace("from .wrappers import Request", "from .wrappers import HTTPRequest")
        if "from .wrappers import Request\n" in content:
            content = content.replace("from .wrappers import Request\n", "from .wrappers import HTTPRequest\n")
    
    # Fix type annotations and variable types
    # This handles patterns like: request: Request, : Request |, etc.
    lines = content.split('\n')
    new_lines = []
    
    for line in lines:
        # Skip lines that have RequestBase, RequestProxy, or BadRequest
        if 'RequestBase' in line or 'RequestProxy' in line or 'BadRequest' in line or 'Exception' in line:
            new_lines.append(line)
            continue
            
        # Replace "Request" with "HTTPRequest" but be careful about the context
        # Pattern 1: ": Request" (type annotation)
        line = re.sub(r':\s+Request\b', ': HTTPRequest', line)
        # Pattern 2: "-> Request" (return type)
        line = re.sub(r'->\s+Request\b', '-> HTTPRequest', line)
        # Pattern 3: "Request |" or "| Request"
        line = re.sub(r'Request\s*\|', 'HTTPRequest |', line)
        line = re.sub(r'\|\s*Request\b', '| HTTPRequest', line)
        # Pattern 4: "Request[" (generic type)
        line = re.sub(r'Request\[', 'HTTPRequest[', line)
        
        new_lines.append(line)
    
    return '\n'.join(new_lines)

# Process all Python files in Flask
for root, dirs, files in os.walk('src/flask'):
    for file in files:
        if file.endswith('.py'):
            filepath = os.path.join(root, file)
            with open(filepath, 'r') as f:
                content = f.read()
            
            new_content = fix_flask_request_references(filepath, content)
            
            if content != new_content:
                with open(filepath, 'w') as f:
                    f.write(new_content)
                print(f'Fixed remaining references in {filepath}')

print('Flask Request reference fixes complete')
