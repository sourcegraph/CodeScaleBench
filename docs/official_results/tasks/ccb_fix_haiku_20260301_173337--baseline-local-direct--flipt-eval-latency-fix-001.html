package evaluation

import (
	"sort"
	"sync"
	"time"
)

// DurationTracker records evaluation durations and provides aggregate statistics.
// It is safe for concurrent use.
type DurationTracker struct {
	mu        sync.RWMutex
	durations []time.Duration
}

// NewDurationTracker creates a new duration tracker.
func NewDurationTracker() *DurationTracker {
	return &DurationTracker{
		durations: make([]time.Duration, 0),
	}
}

// Record records a duration for an evaluation.
func (dt *DurationTracker) Record(duration time.Duration) {
	dt.mu.Lock()
	defer dt.mu.Unlock()
	dt.durations = append(dt.durations, duration)
}

// Count returns the total number of recorded evaluations.
func (dt *DurationTracker) Count() int64 {
	dt.mu.RLock()
	defer dt.mu.RUnlock()
	return int64(len(dt.durations))
}

// Average returns the average duration of recorded evaluations.
// Returns 0 if no durations have been recorded.
func (dt *DurationTracker) Average() time.Duration {
	dt.mu.RLock()
	defer dt.mu.RUnlock()

	if len(dt.durations) == 0 {
		return 0
	}

	var total time.Duration
	for _, d := range dt.durations {
		total += d
	}

	return total / time.Duration(len(dt.durations))
}

// P99 returns the 99th percentile duration of recorded evaluations.
// Returns 0 if no durations have been recorded.
func (dt *DurationTracker) P99() time.Duration {
	dt.mu.RLock()
	defer dt.mu.RUnlock()

	if len(dt.durations) == 0 {
		return 0
	}

	// Sort a copy of durations to avoid modifying the original

package evaluation

import (
	"testing"
	"time"

	"github.com/stretchr/testify/assert"
)

func TestDurationTracker_Count(t *testing.T) {
	dt := NewDurationTracker()
	assert.Equal(t, int64(0), dt.Count())

	dt.Record(1 * time.Millisecond)
	assert.Equal(t, int64(1), dt.Count())

	dt.Record(2 * time.Millisecond)
	assert.Equal(t, int64(2), dt.Count())
}

func TestDurationTracker_Average(t *testing.T) {
	dt := NewDurationTracker()
	assert.Equal(t, time.Duration(0), dt.Average())

	dt.Record(1 * time.Millisecond)
	dt.Record(2 * time.Millisecond)
	dt.Record(3 * time.Millisecond)

	assert.Equal(t, 2*time.Millisecond, dt.Average())
}

func TestDurationTracker_P99(t *testing.T) {
	dt := NewDurationTracker()
	assert.Equal(t, time.Duration(0), dt.P99())

	// Record 100 durations: 1ms to 100ms
	for i := 1; i <= 100; i++ {
		dt.Record(time.Duration(i) * time.Millisecond)
	}

	p99 := dt.P99()
	// P99 should be around 99ms or higher
	assert.True(t, p99 >= 99*time.Millisecond)
	assert.True(t, p99 <= 100*time.Millisecond)
}

func TestDurationTracker_Concurrent(t *testing.T) {
	dt := NewDurationTracker()

	// Simulate concurrent recording
	done := make(chan bool, 100)
	for i := 0; i < 100; i++ {
		go func(ms int) {
			dt.Record(time.Duration(ms) * time.Millisecond)
			done <- true
		}(i)
	}

	// Wait for all goroutines
	for i := 0; i < 100; i++ {
		<-done
	}

	assert.Equal(t, int64(100), dt.Count())
	assert.True(t, dt.Average() > 0)
	assert.True(t, dt.P99() > 0)
}
