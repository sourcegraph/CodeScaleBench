"""
Rate Limiting Middleware.

This module provides a middleware that implements per-IP request rate limiting
using Django's cache framework.
"""

from django.conf import settings
from django.core.cache import cache
from django.http import HttpResponseTooManyRequests
from django.utils.deprecation import MiddlewareMixin


class RateLimitMiddleware(MiddlewareMixin):
    """
    Middleware for rate limiting requests based on client IP address.

    Reads configuration from:
    - settings.RATE_LIMIT_REQUESTS (default: 100)
    - settings.RATE_LIMIT_WINDOW (default: 3600 seconds)
    """

    def __init__(self, get_response):
        super().__init__(get_response)
        self.requests_limit = getattr(settings, "RATE_LIMIT_REQUESTS", 100)
        self.window = getattr(settings, "RATE_LIMIT_WINDOW", 3600)

    def process_request(self, request):
        """
        Check if the client IP has exceeded the rate limit.

        Returns HttpResponseTooManyRequests (429) if limit exceeded,
        otherwise returns None to continue processing.
        """
        # Get client IP from request
        client_ip = request.META.get("REMOTE_ADDR", "")

        if not client_ip:
            return None

        # Create cache key for this IP
        cache_key = f"ratelimit:{client_ip}"

        # Get current request count for this IP
        request_count = cache.get(cache_key, 0)

        # Check if limit exceeded
        if request_count >= self.requests_limit:
            return HttpResponseTooManyRequests()

        # Increment request count
        request_count += 1
        cache.set(cache_key, request_count, self.window)

        return None
