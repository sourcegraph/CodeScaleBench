Http::FilterHeadersStatus CustomHeaderInjectionFilter::decodeHeaders(
    Http::RequestHeaderMap& headers, bool) {
  // Inject request headers based on rules
  if (config_) {
    for (const auto& rule : config_->getRules()) {
      if (rule.inject_on_request) {
        headers.addReference(Http::LowerCaseString(rule.header_name), rule.header_value);
      }
    }
  }
  return Http::FilterHeadersStatus::Continue;
}
