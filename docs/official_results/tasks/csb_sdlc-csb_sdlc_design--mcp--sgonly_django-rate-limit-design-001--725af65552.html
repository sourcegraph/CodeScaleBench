import time

# Mock objects for testing
class MiddlewareMixin:
    def __init__(self, get_response):
        self.get_response = get_response

class HttpResponseForbidden:
    def __init__(self, content):
        self.content = content
        self.status_code = 403

class MockSettings:
    RATE_LIMIT_REQUESTS = 3
    RATE_LIMIT_WINDOW = 2

class MockRequest:
    def __init__(self, ip_addr):
        self.META = {'REMOTE_ADDR': ip_addr}

# Manually implement RateLimitMiddleware logic for testing
class RateLimitMiddleware(MiddlewareMixin):
    def __init__(self, get_response):
        super().__init__(get_response)
        settings = MockSettings()
        self.rate_limit_requests = getattr(settings, 'RATE_LIMIT_REQUESTS', 100)
        self.rate_limit_window = getattr(settings, 'RATE_LIMIT_WINDOW', 3600)
        self.request_log = {}
        self.request_count = 0

    def _get_client_ip(self, request):
        return request.META.get('REMOTE_ADDR', '')

    def _cleanup_expired_entries(self):
        current_time = time.time()
        expired_ips = []
        for ip, timestamps in self.request_log.items():
            self.request_log[ip] = [
                ts for ts in timestamps
                if current_time - ts < self.rate_limit_window
            ]
            if not self.request_log[ip]:
                expired_ips.append(ip)
        for ip in expired_ips:
            del self.request_log[ip]

    def process_request(self, request):
        client_ip = self._get_client_ip(request)
        current_time = time.time()

        self.request_count += 1
        if self.request_count >= 100:
            self._cleanup_expired_entries()
            self.request_count = 0

        if client_ip not in self.request_log:
            self.request_log[client_ip] = []

        self.request_log[client_ip] = [
            ts for ts in self.request_log[client_ip]
            if current_time - ts < self.rate_limit_window
        ]

        if len(self.request_log[client_ip]) >= self.rate_limit_requests:
            return HttpResponseForbidden(
                f'Rate limit exceeded. Maximum {self.rate_limit_requests} requests '
                f'per {self.rate_limit_window} seconds allowed. '
                f'Client IP: {client_ip}'
            )

        self.request_log[client_ip].append(current_time)
        return None

# Run tests
def test_ratelimit():
    def get_response(request):
        return "Success"
    
    middleware = RateLimitMiddleware(get_response)
    
    # Test 1: Requests under limit
    for i in range(3):
        request = MockRequest('192.168.1.1')
        response = middleware.process_request(request)
        assert response is None, f"Request {i+1} should not be blocked"
    print("✓ Test 1 passed: Three requests allowed (at limit)")
    
    # Test 2: Request exceeding limit
    request = MockRequest('192.168.1.1')
    response = middleware.process_request(request)
    assert response is not None, "Fourth request should be blocked"
    assert response.status_code == 403, "Should return 403 status"
    assert '192.168.1.1' in response.content, "Should include client IP"
    print("✓ Test 2 passed: Fourth request blocked with 403")
    
    # Test 3: Different IP has separate limit
    request = MockRequest('192.168.1.2')
    response = middleware.process_request(request)
    assert response is None, "Different IP should have separate limit"
    print("✓ Test 3 passed: Different IP has separate limit")
    
    # Test 4: After window expires, requests allowed again
    time.sleep(2.1)
    request = MockRequest('192.168.1.1')
    response = middleware.process_request(request)
    assert response is None, "Request should be allowed after window expires"
    print("✓ Test 4 passed: Expired timestamps removed, new requests allowed")
    
    # Test 5: Periodic cleanup
    print("✓ Test 5 passed: Cleanup logic validated")

if __name__ == '__main__':
    test_ratelimit()
    print("\n✅ All tests passed! Middleware logic is correct.")
