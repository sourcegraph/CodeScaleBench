// TestTransformerInterfaceContract verifies that transformers implement the basic
// contract of being able to round-trip data through TransformToStorage and TransformFromStorage.
func TestTransformerInterfaceContract(t *testing.T) {
	tests := []struct {
		name        string
		transformer Transformer
		original    []byte
		encrypted   []byte
		expectErr   bool
	}{
		{
			name: "successful round-trip",
			transformer: &mockTransformer{
				encryptResult: []byte("encrypted"),
				decryptResult: []byte("original"),
			},
			original:  []byte("original"),
			encrypted: []byte("encrypted"),
			expectErr: false,
		},
		{
			name: "encryption error",
			transformer: &mockTransformer{
				encryptErr: fmt.Errorf("encryption failed"),
			},
			original:  []byte("original"),
			expectErr: true,
		},
		{
			name: "decryption error",
			transformer: &mockTransformer{
				encryptResult: []byte("encrypted"),
				decryptErr:    fmt.Errorf("decryption failed"),
			},
			original:  []byte("original"),
			encrypted: []byte("encrypted"),
			expectErr: true,
		},
	}

	for _, tt := range tests {
		t.Run(tt.name, func(t *testing.T) {
			ctx := context.Background()

			// Transform to storage
			encrypted, err := tt.transformer.TransformToStorage(ctx, tt.original, nil)
			if (err != nil) != tt.expectErr {
				t.Fatalf("TransformToStorage: unexpected error: %v", err)
			}
			if err != nil {
				return
			}
			if !bytes.Equal(encrypted, tt.encrypted) {
				t.Errorf("TransformToStorage: got %q, want %q", encrypted, tt.encrypted)
			}

			// Transform from storage
			decrypted, stale, err := tt.transformer.TransformFromStorage(ctx, tt.encrypted, nil)
			if (err != nil) != tt.expectErr {
				t.Fatalf("TransformFromStorage: unexpected error: %v", err)
			}
			if err != nil {
				return
			}
			if !bytes.Equal(decrypted, tt.original) {
				t.Errorf("TransformFromStorage: got %q, want %q", decrypted, tt.original)
			}
			if stale {
				t.Errorf("TransformFromStorage: unexpected stale flag")
			}
		})
	}
}

// TestLargeDataTransformation tests transformer behavior with large data payloads.
func TestLargeDataTransformation(t *testing.T) {
	largeData := make([]byte, 1024*1024) // 1MB of data
	for i := range largeData {
		largeData[i] = byte(i % 256)
	}

	mock := &mockTransformer{
		encryptResult: largeData,
		decryptResult: largeData,
	}

	pt := NewPrefixTransformers(
		fmt.Errorf("no match"),
		PrefixTransformer{Prefix: []byte("large:"), Transformer: mock},
	)

	// Test encryption of large data
	result, err := pt.TransformToStorage(context.Background(), largeData, nil)
	if err != nil {
		t.Fatalf("TransformToStorage failed: %v", err)
	}
	if len(result) != len(largeData)+6 { // 6 bytes for "large:" prefix
		t.Errorf("result size mismatch: got %d, want %d", len(result), len(largeData)+6)
	}

	// Test decryption of large data
	dataWithPrefix := append([]byte("large:"), largeData...)
	decrypted, _, err := pt.TransformFromStorage(context.Background(), dataWithPrefix, nil)
	if err != nil {
		t.Fatalf("TransformFromStorage failed: %v", err)
	}
	if !bytes.Equal(decrypted, largeData) {
		t.Error("decrypted data doesn't match original")
	}
}
