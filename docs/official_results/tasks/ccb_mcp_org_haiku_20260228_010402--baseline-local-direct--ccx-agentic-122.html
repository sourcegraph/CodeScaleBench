  Shutdown();
}

nsresult SampleObserverComponent::Initialize() {
  // Get the global observer service
  mObserverService = mozilla::services::GetObserverService();
  if (!mObserverService) {
    return NS_ERROR_FAILURE;
  }

  return NS_OK;
}

void SampleObserverComponent::Shutdown() {
  if (mIsShuttingDown) {
    return;
  }

  mIsShuttingDown = true;
  mObserverService = nullptr;
}
